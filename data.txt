		(() => {
			const RESET = 10;
			let timer = RESET;

			const refresh = async () => {
				const response = await fetch(`data.txt?version=${Date.now()}`);
				const highlights = await response.text();
				document.querySelector(".highlights").innerHTML = highlights;
				timer = RESET;
			};


			setInterval(() => {
				timer -= 1;
				if (timer <= 0) refresh();
				document.querySelector(".update span").innerText = timer;
			}, 1000);
		})();
